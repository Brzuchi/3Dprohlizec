async function fetchSharedModels() { const { data, error } = await supabase.storage.from('models').list(`shared/${urlId}`); if (error || !data || data.length === 0) { emptyMsg.innerText = "Sestava nebyla nalezena nebo je prázdná."; hideLoader(); return; } // --- PALETA BAREV --- // Přidali jsme různé kontrastní barvy. Pokud bude dílů víc než barev, pojedou zase od začátku. const colors = [ 0x007bff, // Modrá 0xdc3545, // Červená 0x28a745, // Zelená 0xffc107, // Žlutá 0x17a2b8, // Tyrkysová 0x6f42c1, // Fialová 0xfd7e14, // Oranžová 0xe83e8c // Růžová ]; let colorIndex = 0; for (const file of data) { if(file.name === '.emptyFolderPlaceholder') continue; const { data: publicUrlData } = supabase.storage.from('models').getPublicUrl(`shared/${urlId}/${file.name}`); // Přiřadíme barvu z palety const modelColor = colors[colorIndex % colors.length]; await loadAndRender(publicUrlData.publicUrl, file.name, modelColor); colorIndex++; // Posuneme se na další barvu pro další díl } focusCamera(); uploadBox.style.display = 'none'; resetBtn.style.display = 'block'; hideLoader(); }
